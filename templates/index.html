function showPaymentModal(productId, productName) { currentProductId = productId; document.getElementById('mTitle').innerText = productName; document.getElementById('mBody').innerHTML = `

Pay Anonymously with Crypto (USDT)

The Exact KES amount will automatically be converted to USDT. You will be redirected securely to the Binance Checkout Portal.

`; let confirmBtn = document.getElementById('mConfirm'); confirmBtn.style.display = 'block'; confirmBtn.innerText = "PROCEED TO M-PESA CHECKOUT"; confirmBtn.style.backgroundColor = 'var(--navy)'; confirmBtn.style.color = 'white'; confirmBtn.onclick = processPaymentRoute; document.getElementById('global-modal').style.display = 'flex'; } function handleNetworkChange() { let network = document.getElementById('networkSelect').value; document.getElementById('safaricom-flow').style.display = network === 'safaricom' ? 'block' : 'none'; document.getElementById('binance-flow').style.display = network === 'binance' ? 'block' : 'none'; let btn = document.getElementById('mConfirm'); if (network === 'binance') { btn.style.backgroundColor = '#fcd535'; btn.style.color = 'black'; btn.innerText = 'PAY WITH BINANCE (USDT)'; } else { btn.style.backgroundColor = 'var(--navy)'; btn.style.color = 'white'; btn.innerText = 'PROCEED TO M-PESA CHECKOUT'; } } function processPaymentRoute() { let network = document.getElementById('networkSelect').value; if (network === 'safaricom') { let phone = document.getElementById('mInput').value; if(!phone) return alert('Safaricom number required.'); triggerSTKPush(phone); } else if (network === 'binance') { triggerBinancePay(); } } async function triggerBinancePay() { document.getElementById('mConfirm').style.display = 'none'; document.getElementById('mBody').innerHTML = `

GENERATING SECURE BINANCE PORTAL...

`; try { let res = await fetch('/binance-pay', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({product_id: currentProductId}) }); let data = await res.json(); if(data.status === 'success') { // Start polling the server to wait for Binance payment confirmation in the background pollPayment(data.checkout_id); // Open Binance Pay portal in a new tab window.open(data.checkout_url, '_blank'); document.getElementById('mBody').innerHTML += `

Awaiting Binance confirmation on the blockchain. Do not close this window.

`; } else { document.getElementById('mBody').innerHTML = `

ERROR: ${data.error}

`; } } catch(e) { document.getElementById('mBody').innerHTML = `

Network Error connecting to Binance.

`; } }
